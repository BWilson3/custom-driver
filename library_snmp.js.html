var snmp = require("net-snmp");

/**
 * Creates an SNMP session for your device
 * @constructor
 * @param {Object} device    - The Device object
 */
function createSessionForDevice(device) {
    /** 
     * Session initialization
     * @private 
     * @readonly
    */
    var realSession = snmp.createSession(device.ip, "public");
    return {
        /** @property {function} walk            - Initiates an SNMP Walk towards the device */
        walk: function () {
            return realSession.walk.apply(realSession, arguments);
        },
        /** @property {function} get            - Initiates an SNMP Get towards the device */
        get: function () {
            return realSession.get.apply(realSession, arguments);
        },
        /** 
         * @property {function} isVarbindError  - Checks if the provided variable binding has an error
         * @param {Object} varbind              - The Variable Binding object
         * @return {boolean}                    - The check result
        */
        isVarbindError: snmp.isVarbindError,
    };
}


module.exports.createSessionForDevice = createSessionForDevice;
