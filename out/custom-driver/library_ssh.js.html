var async = require('async');
var events = require('events')
var ssh2 = require('ssh2');
var q = require('q');
var lodash = require('lodash');

var sandboxConsole;

/**
 * SSH Library Resource Locator object 
 * @private
 * @readonly
 */
var sandboxResourceLocator = {
    log: {
        decorateLogs: function (){
            return sandboxConsole;
        }
    },
    ssh2: ssh2,
    q: q,
    async: async,
    lodash: lodash,
    events: events,
}
/**
 * Default SSH Options algorithms parameter
 * @private
 * @constant {Object} 
 * @property {Array}  cipher      - Default algorithm ciphers
 * @property {Array}  kex         - Default algorithm kex
*/
const DEFAULT_ALGORITHMS = {
    cipher: [
        'aes128-gcm',
        'aes128-gcm@openssh.com',
        'aes256-gcm',
        'aes256-gcm@openssh.com'
    ],
    kex: [
        'diffie-hellman-group-exchange-sha1',
        'diffie-hellman-group-exchange-sha256',
        'ecdh-sha2-nistp256',
        'ecdh-sha2-nistp384',
        'ecdh-sha2-nistp521'
    ]

}

/**
 * Sends a sequence of SSH commands
 * @function
 * @private
 * @readonly
 * @param {Object} myConsole      - The Domotz Sandbox console
 * @param {SshOptions} options    - The SSH Commands exection options
*/
function sendSSHShellSequence(myConsole, options, callback) {
    sandboxConsole = myConsole
    options.algorithms = options.algorithms || DEFAULT_ALGORITHMS
    function outputTransformer(result){
        var error = result.error
        var output = [];
        if (result.output) {
            result.output.forEach(function (item) {
                output.push(Buffer(item, 'base64').toString());
            })
        }
        callback(output, error)
    }
    var sshNode = require('../../network_tools/sshNode').factory(sandboxResourceLocator);
    return sshNode.sshShellSequence(options, outputTransformer);
}

module.exports.sendSSHShellSequence = sendSSHShellSequence;
