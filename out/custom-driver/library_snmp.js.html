/**

 * @namespace snmpSession
*/
/**
 * The SNMP library Options
 * @typedef  {Object} SnmpOptions
 */

var snmp = require("net-snmp");

/**
 * Creates an SNMP session for your device
 * @constructor
 * @private
 * @param {Device} device    - The Device object
 */
function createSessionForDevice(device) {
    /** 
     * Session initialization
     * @private 
     * @readonly
    */
    var realSession = snmp.createSession(device.ip, "public");
    return {
        /**
         * Initiates an SNMP Walk towards the device
         * Usage: snmpSession.walk(options)
         * @memberof snmpSession
         * @readonly
         * @param  {SnmpOptions} options - The SNMP Walk Options
         * @function
         * @return    {any}              - The SNMP Walk result
         */
        walk: function (options) {
            return realSession.walk.apply(realSession, options);
        },
        /**
         * Initiates an SNMP Get towards the device
         * Usage: snmpSession.get(options)
         * @memberof snmpSession
         * @readonly
         * @param    {SnmpOptions} options  - The SNMP Walk Options
         * @function
         * @return    {any}                 - The SNMP Walk result
         */        
        get: function (options) {
            return realSession.get.apply(realSession, options);
        },
        /** 
         * Checks if the provided variable binding has an error
         * Usage: snmpSession.isVarbindError(varbind)
         * @memberof snmpSession
         * @readonly
         * @function 
         * @param {Object} varbind              - The Variable Binding object
         * @return {boolean}                    - Returns True if there is a Variable binding error and false otherwise
        */
        isVarbindError: snmp.isVarbindError,
    };
}


module.exports.createSessionForDevice = createSessionForDevice;
